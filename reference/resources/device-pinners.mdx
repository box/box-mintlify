<ParamField body="entries" type="Device pinnerarray">

</ParamField>

<ParamField body="limit" type="integer(int64)">

`limit` query parameter unless that value exceeded the maximum value
allowed.
</ParamField>

<ParamField body="next_marker" type="integer(int64)">

The marker for the start of the next page of results.
</ParamField>

<ParamField body="order" type="object[]">

<Expandable>
  <ParamField body="by" type="string">

  Value is always `id`
  </ParamField>

  <ParamField body="direction" type="string">
  example: `asc`

  Value is one of `asc`,`desc`
  </ParamField>

</Expandable>

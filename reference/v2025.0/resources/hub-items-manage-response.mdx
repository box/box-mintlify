Response schema for the status of Box Hub items management operations.

  <ParamField body="action" type="string">
  example: `add`

  </ParamField>

  <ParamField body="error" type="string">
  example: `Item not found`
  
  Error message if the operation failed.
  </ParamField>

  <ParamField body="item" type="objectobject">
  Reference to an item that can be added to a Box Hub.
  </ParamField>

  <ParamField body="status" type="integer">
  example: `200`
  
  The HTTP status code of the operation.
  </ParamField>

</Expandable>
</ParamField>

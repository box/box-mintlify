ファイルまたはフォルダに適用されているメタデータインスタンスのリスト。

<ParamField body="entries" type="メタデータインスタンスarray">

</ParamField>

<ParamField body="limit" type="integer">

</ParamField>

```json レスポンスの例
      "$scope": "enterprise_27335",
      "$template": "marketingCollateral",
      "$version": 1
    }
  ],
  "limit": 100
}
```

</ResponseExample>
